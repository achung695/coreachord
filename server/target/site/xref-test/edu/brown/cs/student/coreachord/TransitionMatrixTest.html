144   @Test
145   public void testDigitize() {
146     int counter = 0;
147     while (counter < 1000) {
148       List<Tuple<Double, Integer>> l1 = new ArrayList<>();
149       l1.add(new Tuple<>(.2, 5));
150       l1.add(new Tuple<>(.5, 6));
151       l1.add(new Tuple<>(.8, 7));
152       l1.add(new Tuple<>(1., 8));
153       int[] buckets1 = new int[4];
154       List<Tuple<Double, Integer>> l2 = new ArrayList<>();
155       l2.add(new Tuple<>(.1, 0));
156       l2.add(new Tuple<>(.5, 1));
157       l2.add(new Tuple<>(.8, 2));
158       int[] buckets2 = new int[3];
159       List<Tuple<Double, Integer>> l3 = new ArrayList<>();
160       l3.add(new Tuple<>(1.5, 7));
161       l3.add(new Tuple<>(3.9, 8));
162       l3.add(new Tuple<>(5.6, 9));
163       l3.add(new Tuple<>(5.7, 10));
164       l3.add(new Tuple<>(6.0, 11));
165       l3.add(new Tuple<>(10.0, 12));
166       int[] buckets3 = new int[6];
167       for (int i = 0; i < 100000; i++) {
168         buckets1[tmat1.digitize(l1, 1.0) - 5] += 1;
169         buckets2[tmat2.digitize(l2, 0.8)] += 1;
170         buckets3[tmat3.digitize(l3, 10.0) - 7] += 1;
171       }
172       assertEquals(buckets1[0], 20000, 2000); //.2
173       assertEquals(buckets1[1], 30000, 3000); //.3
174       assertEquals(buckets1[2], 30000, 3000); //.3
175       assertEquals(buckets1[3], 20000, 2000); //.2
176 
177       assertEquals(buckets2[0], 12500, 1250); //.125
178       assertEquals(buckets2[1], 50000, 5000); //.5
179       assertEquals(buckets2[2], 35000, 3500); //.375
180 
181       assertEquals(buckets3[0], 15000, 1500); //.15
182       assertEquals(buckets3[1], 24000, 2400); //.24
183       assertEquals(buckets3[2], 17000, 1700); //.17
184       assertEquals(buckets3[3], 1000, 150); //.01
185       assertEquals(buckets3[4], 3000, 300); //.03
186       assertEquals(buckets3[5], 40000, 4000); //.4
187       counter++;
188     }
189     assertEquals(tmat1.digitize(new ArrayList<>(), 0), -1);
190   }
