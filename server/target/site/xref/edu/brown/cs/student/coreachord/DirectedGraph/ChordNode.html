8   public class ChordNode implements IGraphNode {
9     private final Chord chord;
10    private List<IEdge> outgoingedges;
11    public ChordNode(Chord inputchord) {
12      chord = inputchord;
13      outgoingedges = new ArrayList<>();
14    }
15    @Override
16    public List<IEdge> getEdges() {
17      return outgoingedges;
18    }
19    @Override
20    public void addEdge(IGraphNode target, double weight) {
21      outgoingedges.add(new Edge(this, target, weight));
22    }
23    @Override
24    public Chord getNode() {
25      return chord;
26    }
27    @Override
28    public IEdge findEdgeTo(IGraphNode node) {
29      for (IEdge e : outgoingedges) {
30        if (e.to().equals(node)) {
31          return e;
32        }
33      }
34      return null;
35    }
36  }
