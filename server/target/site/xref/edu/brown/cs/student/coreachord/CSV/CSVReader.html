3   import java.io.BufferedReader;
4   import java.io.FileReader;
5   import java.util.ArrayList;
6   import java.util.List;
7   
8   /**
9    * Reads in a CSV file and outputs a matrix.
10   */
11  public class CSVReader {
12  
13    /**.
14     * convert a csv file to a list of list of strings
15     * @param filepath - path to the csv file to read
16     * @return - list of list of strings of raw CSV data
17     */
18    public List<String[]> parseCSV(String filepath) {
19      List<String[]> csvList = new ArrayList<>();
20  
21      try {
22        BufferedReader csvReader = new BufferedReader(new FileReader(filepath));
23  
24        String line = csvReader.readLine();
25  
26        while (line != null) {
27          // get string array from csv split
28          String[] split = line.split(",");
29          csvList.add(split);
30          line = csvReader.readLine();
31        }
32  
33        csvReader.close();
34  
35      } catch (Exception e) {
36        System.err.println("Invalid CSV");
37      }
38      return csvList;
39    }
40  
41  }
