3   /*
4    * A wrapper class of Chord.
5    * It's basically a Chord, but it also
6    * contains information about chord progression as well.
7    * (like length of the chord, etc.)
8    */
9   public class GeneratedChord {
10    private Chord chorddata;
11    private int chordlength;
12    public GeneratedChord(Chord chord, int length) {
13      chorddata = chord;
14      chordlength = length;
15    }
16  
17    public Chord getChorddata() {
18      return chorddata;
19    }
20    public int getChordlength() {
21      return chordlength;
22    }
23    @Override
24    public String toString() {
25      return chorddata + "\nChord Length: " + chordlength;
26    }
27  }
