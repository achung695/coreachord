20    /*
21     * Method for filling transition matrix.
22     * It'll take in a list of array of strings that contain
23     * probability data for each row of the transition matrix.
24     * The probabilities list of string arrays come from
25     * csv parsing.
26     * @param
27     */
28    private void fillTransitionMatrix(List<String[]> probabilities) {
29      // get lengths of dimensions of probability matrix
30      int horizontallen = probabilities.get(0).length;
31      int verticallen = probabilities.size();
32      // initialize transition matrix
33      this.tmat = new double[verticallen - 1][horizontallen - 1];
34  
35      for (int i = 1; i < verticallen; i++) {
36        for (int j = 1; j < horizontallen; j++) {
37          // get individual probability
38          String currprobstring = probabilities.get(i)[j];
39          // store into each index of tmat
40          this.tmat[i - 1][j - 1] = Double.parseDouble(currprobstring);
41        }
42      }
43    }

46    /**
47     * Helper method that gives you corresponding chord
48     * based on the index in transition matrix.
49     * (Handles chord-index correspondence in the transition
50     * matrix)
51     *
52     * @param index
53     * @param numqualities - Number of qualities a chord can have
54     * @return corresponding Chord
55     */
56    public static Chord getCorrespondingChord(int index, int numqualities) {
57      // Root order: C, Db, D, Eb, E, F, Gb, G, Ab, A, Bb, B
58      // Quality order: MAJOR7, MINOR7, MINOR7FLAT5, DOMINANT7
59      int rootordinal = index / numqualities; // integer division to get root ordinal
60      int qualityordinal = index % numqualities; // mod to get quality ordinal (get remainder)
61      CoreaApplication.Root root = CoreaApplication.Root.values()[rootordinal];
62      CoreaApplication.Quality quality = CoreaApplication.Quality.values()[qualityordinal];
63      return new Chord(root, quality);
64    }

66    /*
67     * Explanation:
68     *
69     * This selection method is a helper method that selects the
70     * next chord to go to (selects the column in the transition matrix)
71     * while taking into account the probability distribution for a given row.
72     *
73     * The way this method is implemented is that it makes use of the concept of
74     * "cumulative probability distribution" in probability theory, which
75     * weighs the higher probability index more than the other ones.
76     *
77     * We first initialize a cumulative probability distribution array,
78     * populate the array with the current row's probability information,
79     * and index into that array using a randomly initialized double in [0,1)
80     * (the result of this search will be our newly selected chord value).
81     *
82     * Then, we look for that probability value in our original row in transition matrix.
83     * The column index that contains that probability value will be indicating
84     * our next chord to go to.
85     */
86    public int getNextChordIndex(int row, int numQualities) {
87      if (this.tmat == null) {
88        return -1;
89      }
90  
91      Chord currchord = TransitionMatrix.getCorrespondingChord(row, numQualities);
92      // cumulative probability distribution matrix
93      int rowlen = this.tmat[0].length; // row's length
94      double[] currrow = this.tmat[row]; // get current row (current chord).
95  
96      List<Tuple<Double, Integer>> cumulativeDist = new ArrayList<>();
97  
98      double cumulative = 0;
99      for (int i = 0; i < rowlen; i++) {
100       double curr = tmat[row][i];
101       // skips zero probabilities
102       if (curr > 0) {
103         cumulative += curr; // sum up as we iterate through the current row.
104         // double cumulative distribution to row index
105         cumulativeDist.add(new Tuple<>(cumulative, i));
106       }
107     }
108 
109     // get index from probability distribution
110     return digitize(cumulativeDist, 1.0);
111   }

113   /**
114    * binning with a cumulative distribution array.
115    *
116    * @param cumulativeDist - monotonically increasing array where the final
117    *                       value should equal 1, representing cumulative distribution
118    * @param sumOfProbabbilities -
119    * @return - the random index given the weights of the array
120    */
121   public int digitize(List<Tuple<Double, Integer>> cumulativeDist, double sumOfProbabbilities) {
122     double rand = Math.random() * sumOfProbabbilities;
123 
124     for (Tuple<Double, Integer> probToIndex : cumulativeDist) {
125       // if generated value is less than the probability cumulative distribution
126       if (rand <= probToIndex.getField1()) {
127         // return index of that
128         return probToIndex.getField2();
129       }
130     }
131     // error, we reached the end of the list without return indices
132     return -1;
133   }
